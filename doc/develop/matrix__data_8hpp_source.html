393  static matrix_data cond(size_type size,
394  remove_complex<ValueType> condition_number,
395  RandomDistribution &&dist, RandomEngine &&engine,
396  size_type num_reflectors)
397  {
398  using range = range<accessor::row_major<ValueType, 2>>;
399  std::vector<ValueType> mtx_data(size * size, zero<ValueType>());
400  std::vector<ValueType> ref_data(size);

403  range reflector(ref_data.data(), size, 1u, 1u);

405  initialize_diag_with_cond(condition_number, matrix);
406  for (size_type i = 0; i < num_reflectors; ++i) {
407  generate_random_reflector(dist, engine, reflector);

472  template <typename Accessor>
473  static void initialize_diag_with_cond(
474  remove_complex<ValueType> condition_number,
475  const range<Accessor> &matrix)
476  {
477  using sigma_type = remove_complex<ValueType>;
478  const auto size = matrix.length(0);
479  const auto min_sigma = one(condition_number) / sqrt(condition_number);
480  const auto max_sigma = sqrt(condition_number);
481 
482  matrix = zero(matrix);
483  for (gko::size_type i = 0; i < size; ++i) {
484  matrix(i, i) = max_sigma * static_cast<sigma_type>(size - i - 1) /
485  static_cast<sigma_type>(size - 1) +
486  min_sigma * static_cast<sigma_type>(i) /
487  static_cast<sigma_type>(size - 1);
488  }
489  }

502  template <typename Accessor>
503  static void reflect_domain(const range<Accessor> &reflector,
504  const range<Accessor> &matrix,
505  ValueType *work_data)
506  {
507  const auto two = one<ValueType>() + one<ValueType>();
508  range<accessor::row_major<ValueType, 2>> work(work_data,
509  matrix.length(0), 1u, 1u);
510  work = mmul(matrix, reflector);
511  const auto ct_reflector = conj(transpose(reflector));
512  const auto scale = two / mmul(ct_reflector, reflector)(0, 0);
513  matrix = matrix - scale * mmul(work, ct_reflector);
514  }

516  template <typename Accessor>
517  static void reflect_range(const range<Accessor> &reflector,
518  const range<Accessor> &matrix,
519  ValueType *work_data)
520  {
521  const auto two = one<ValueType>() + one<ValueType>();
522  range<accessor::row_major<ValueType, 2>> work(
523  work_data, 1u, matrix.length(0), matrix.length(0));
524  const auto ct_reflector = conj(transpose(reflector));
525  work = mmul(ct_reflector, matrix);
526  const auto scale = two / mmul(ct_reflector, reflector)(0, 0);
527  matrix = matrix - scale * mmul(reflector, work);
