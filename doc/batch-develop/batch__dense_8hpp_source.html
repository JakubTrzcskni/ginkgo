366  inline size_type compute_batch_mem(const batch_dim<2>& sizes,
367  const batch_stride& strides)
368  {
369  GKO_ASSERT(sizes.get_num_batch_entries() ==
370  strides.get_num_batch_entries());
371  if (sizes.stores_equal_sizes() && strides.stores_equal_strides()) {
372  return (sizes.at(0))[0] * strides.at(0) *
373  sizes.get_num_batch_entries();
374  }
375  size_type mem_req = 0;
376  for (auto i = 0; i < sizes.get_num_batch_entries(); ++i) {
377  mem_req += (sizes.at(i))[0] * strides.at(i);
378  }
379  return mem_req;
380  }

385  inline batch_stride extract_nth_dim(const int dim, const batch_dim<2>& size)
386  {
387  if (size.stores_equal_sizes()) {
388  return batch_stride(size.get_num_batch_entries(), size.at(0)[dim]);
389  }
390  std::vector<size_type> stride(size.get_num_batch_entries());
391  for (auto i = 0; i < size.get_num_batch_entries(); ++i) {
392  stride[i] = (size.at(i))[dim];
393  }
394  return batch_stride(stride);

427  inline Array<size_type> compute_num_elems_per_batch_cumul(
428  std::shared_ptr<const Executor> exec, const batch_dim<2>& sizes,
429  const batch_stride& strides)
430  {
431  auto num_elems = Array<size_type>(exec->get_master(),
432  sizes.get_num_batch_entries() + 1);
433  num_elems.get_data()[0] = 0;
434  for (auto i = 0; i < sizes.get_num_batch_entries(); ++i) {
435  num_elems.get_data()[i + 1] =
436  num_elems.get_data()[i] + (sizes.at(i))[0] * strides.at(i);
437  }
438  num_elems.set_executor(exec);
439  return num_elems;
440  }

686 template <typename Matrix, typename... TArgs>
687 std::unique_ptr<Matrix> batch_initialize(
688  std::vector<size_type> stride,
689  std::initializer_list<std::initializer_list<typename Matrix::value_type>>
690  vals,
691  std::shared_ptr<const Executor> exec, TArgs&&... create_args)
692 {
693  using batch_dense = matrix::BatchDense<typename Matrix::value_type>;
694  size_type num_batch_entries = vals.size();
695  std::vector<size_type> num_rows(num_batch_entries);
696  std::vector<dim<2>> sizes(num_batch_entries);
697  auto vals_begin = begin(vals);
698  for (size_type b = 0; b < num_batch_entries; ++b) {
699  num_rows[b] = vals_begin->size();
700  sizes[b] = dim<2>(num_rows[b], 1);
701  vals_begin++;
702  }
703  auto b_size = batch_dim<2>(sizes);
704  auto b_stride = batch_stride(stride);
705  auto tmp = batch_dense::create(exec->get_master(), b_size, b_stride);
706  size_type batch = 0;
707  for (const auto& b : vals) {
708  size_type idx = 0;
709  for (const auto& elem : b) {
710  tmp->at(batch, idx) = elem;
711  ++idx;
712  }
713  ++batch;
714  }
715  auto mtx = Matrix::create(exec, std::forward<TArgs>(create_args)...);
716  tmp->move_to(mtx.get());
717  return mtx;
718 }
