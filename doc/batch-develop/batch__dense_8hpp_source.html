488  inline size_type compute_batch_mem(const batch_dim<2> &sizes,
489  const batch_stride &strides)
490  {
491  GKO_ASSERT(sizes.get_num_batch_entries() ==
492  strides.get_num_batch_entries());
493  if (sizes.stores_equal_sizes() && strides.stores_equal_strides()) {
494  return (sizes.at(0))[0] * strides.at(0) *
495  sizes.get_num_batch_entries();
496  }
497  size_type mem_req = 0;
498  for (auto i = 0; i < sizes.get_num_batch_entries(); ++i) {
499  mem_req += (sizes.at(i))[0] * strides.at(i);
500  }
501  return mem_req;
502  }

507  inline batch_stride extract_nth_dim(const int dim, const batch_dim<2> &size)

549  inline Array<size_type> compute_num_elems_per_batch_cumul(
550  std::shared_ptr<const Executor> exec, const batch_dim<2> &sizes,
551  const batch_stride &strides)
552  {
553  auto num_elems = Array<size_type>(exec->get_master(),
554  sizes.get_num_batch_entries() + 1);
555  num_elems.get_data()[0] = 0;
556  for (auto i = 0; i < sizes.get_num_batch_entries(); ++i) {
557  num_elems.get_data()[i + 1] =
558  num_elems.get_data()[i] + (sizes.at(i))[0] * strides.at(i);
559  }
560  num_elems.set_executor(exec);
561  return num_elems;
562  }

808 template <typename Matrix, typename... TArgs>
809 std::unique_ptr<Matrix> batch_initialize(
810  std::vector<size_type> stride,
811  std::initializer_list<std::initializer_list<typename Matrix::value_type>>
812  vals,
813  std::shared_ptr<const Executor> exec, TArgs &&... create_args)
814 {
815  using batch_dense = matrix::BatchDense<typename Matrix::value_type>;
816  size_type num_batch_entries = vals.size();
817  std::vector<size_type> num_rows(num_batch_entries);
818  std::vector<dim<2>> sizes(num_batch_entries);
819  auto vals_begin = begin(vals);
820  for (size_type b = 0; b < num_batch_entries; ++b) {
821  num_rows[b] = vals_begin->size();
822  sizes[b] = dim<2>(num_rows[b], 1);
823  vals_begin++;
824  }
825  auto b_size = batch_dim<2>(sizes);
826  auto b_stride = batch_stride(stride);
827  auto tmp = batch_dense::create(exec->get_master(), b_size, b_stride);
828  size_type batch = 0;
829  for (const auto &b : vals) {
830  size_type idx = 0;
831  for (const auto &elem : b) {
832  tmp->at(batch, idx) = elem;
833  ++idx;
834  }
835  ++batch;
836  }
837  auto mtx = Matrix::create(exec, std::forward<TArgs>(create_args)...);
838  tmp->move_to(mtx.get());
839  return mtx;
840 }
