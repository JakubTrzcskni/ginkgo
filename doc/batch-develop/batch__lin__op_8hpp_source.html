553 template <typename ValueType>
554 class EnableBatchScaling : public BatchScalable {
555 public:
565  void batch_scale(const BatchLinOp *const left_scale,
566  const BatchLinOp *const right_scale) override
567  {
568  if (!left_scale && !right_scale) {
569  return;
570  }
571 
572  if (left_scale) {
573  if (!dynamic_cast<const matrix::BatchDense<ValueType> *>(
574  left_scale))
575  GKO_NOT_SUPPORTED(left_scale);
576  }
577  if (right_scale) {
578  if (!dynamic_cast<const matrix::BatchDense<ValueType> *>(
579  right_scale))
580  GKO_NOT_SUPPORTED(right_scale);
581  }
582 
583  /* TODO: Somehow restrict BatchScalable and EnableBatchScaling to
584  * BatchLinOp so that the following dynamic_cast is not needed.
585  */
586  if (auto blp = dynamic_cast<BatchLinOp *>(this)) {
587  GKO_ASSERT_BATCH_SCALABLE_TWO_SIDED(blp, left_scale, right_scale);
588  } else {
589  GKO_NOT_SUPPORTED(this);
590  }
591 
592  this->batch_scale_impl(left_scale, right_scale);
593  }
594 
595 private:
596  virtual void batch_scale_impl(const BatchLinOp *left_scale,
597  const BatchLinOp *right_scale) = 0;
